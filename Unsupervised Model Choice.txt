import pandas as pd
from sklearn.preprocessing import StandardScaler
from sklearn.ensemble import IsolationForest
from sklearn.neighbors import LocalOutlierFactor
from sklearn.metrics import (
    confusion_matrix, precision_score, recall_score, f1_score, roc_auc_score
)

# 1. Load dataset
df = pd.read_csv("creditcard.csv")
print(df.head())
print(df["Class"].value_counts())  # 1 = fraud, 0 = normal (Kaggle format)

X = df.drop("Class", axis=1)
y = df["Class"]

# 2. Scaling
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 3. Isolation Forest (unsupervised)
iso = IsolationForest(
    n_estimators=200,
    contamination=0.001,   # adjust ~ expected fraud ratio
    random_state=42
)
iso.fit(X_scaled)
# IsolationForest outputs -1 for anomalies, 1 for normal
y_pred_iso = iso.predict(X_scaled)
y_pred_iso = (y_pred_iso == -1).astype(int)  # convert to 1 = fraud, 0 = normal

# 4. Local Outlier Factor (LOF)
lof = LocalOutlierFactor(
    n_neighbors=20,
    contamination=0.001,
    novelty=False   # classic LOF = no separate predict()
)
y_pred_lof = lof.fit_predict(X_scaled)
y_pred_lof = (y_pred_lof == -1).astype(int)

def eval_unsupervised(name, y_true, y_pred, scores=None):
    cm = confusion_matrix(y_true, y_pred)
    prec = precision_score(y_true, y_pred, zero_division=0)
    rec = recall_score(y_true, y_pred, zero_division=0)
    f1 = f1_score(y_true, y_pred, zero_division=0)

    print(f"\n{name}")
    print("Confusion Matrix:\n", cm)
    print(f"Precision: {prec:.4f}")
    print(f"Recall:    {rec:.4f}")
    print(f"F1-score:  {f1:.4f}")

    # If anomaly scores are given, compute ROC-AUC
    if scores is not None:
        auc = roc_auc_score(y_true, scores)
        print(f"ROC-AUC:   {auc:.4f}")

# For IsolationForest we can get scores:
iso_scores = -iso.decision_function(X_scaled)   # higher = more anomalous
eval_unsupervised("Isolation Forest", y, y_pred_iso, scores=iso_scores)

# For LOF, decision_function not available; we can use negative_outlier_factor_
lof_scores = -lof.negative_outlier_factor_
eval_unsupervised("Local Outlier Factor", y, y_pred_lof, scores=lof_scores)
