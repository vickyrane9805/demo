# Assignment 3: Compare Linear Regression vs Logistic Regression (Classification Only Demo)

import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import accuracy_score, f1_score, confusion_matrix
from sklearn.linear_model import LogisticRegression, LinearRegression

# 1. Load dataset
df = pd.read_csv("data.csv")
X = df.drop("target", axis=1)
y = df["target"]

# 2. Preprocessing
X = X.fillna(X.median())

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 3. Train/Test split
X_train, X_test, y_train, y_test = train_test_split(
    X_scaled, y, test_size=0.2, random_state=42, stratify=y
)

# --- Models ---
log_reg = LogisticRegression(max_iter=1000)
lin_reg = LinearRegression()

log_reg.fit(X_train, y_train)
lin_reg.fit(X_train, y_train)

# Predictions
y_pred_log = log_reg.predict(X_test)

# Linear regression prediction â†’ threshold to convert into class
y_pred_lin = lin_reg.predict(X_test)
y_pred_lin = (y_pred_lin >= 0.5).astype(int)

# Evaluation
print("\nLogistic Regression")
print("Accuracy:", accuracy_score(y_test, y_pred_log))
print("F1:", f1_score(y_test, y_pred_log, average='weighted'))
print("Confusion Matrix:\n", confusion_matrix(y_test, y_pred_log))

print("\nLinear Regression (thresholded)")
print("Accuracy:", accuracy_score(y_test, y_pred_lin))
print("F1:", f1_score(y_test, y_pred_lin, average='weighted'))
print("Confusion Matrix:\n", confusion_matrix(y_test, y_pred_lin))
